<?php
namespace App\Mail;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Mail\Mailable;
use Illuminate\Queue\SerializesModels;
class TestMailing extends Mailable
{
    public $title;  //check it out
    use Queueable, SerializesModels;
    public function __construct($title)//check it out
    {
        $this->title = $title;//check it out
    }
    public function build()
    {
        return $this->view('mail.testmailing');//check it out
    }
}

php artisan make:controller TestMailingController

app/Http/Controllers/TestMailingController.php file.

<?php
namespace App\Http\Controllers;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Mail;
use App\Mail\TestMailing;
class TestMailingController extends Controller
{
    public function form(Request $request)
    {
        return view('mail.form');
    }
    public function send(Request $request)
    {
        Mail::to($request->mail)->send(new TestMailing($request->title));
        return 'Mail sent';
    }
}

routes/web.php file.

Route::get('/mail', 'TestMailingController@form')->name('mail');
